\documentclass{article}
% To generate template specific to Supernote Nomad dimensions
% Feel free to adjust the document as per your taste.

% Nomad dimensions
%   Device: 
%       L: 191.85mm, W: 139.2mm
%   Screen: 
%       L: 158mm, W: 119mm
% Toolbar is roughly about 8mm

% Arguments
%   top,bottom,right,left: Define margin
%   showframe: Helps with visualing the margin and can be removed
\usepackage[
  paperheight=158mm,
  paperwidth=119mm,
  top=0mm,
  bottom=0mm,
  right=0mm,
  left=0mm,
  heightrounded,
  showframe
]{geometry}

% Arguments:
%   patternsize: Controls the distance between dots
\usepackage[
  pattern=dot, 
  patternsize=6mm, 
  dotsize=0.2mm,
  textarea
]{gridpapers}

\begin{document}

	~ %otherwise there will be an error

\end{document}
